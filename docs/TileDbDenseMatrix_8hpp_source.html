39 TileDbDenseMatrix(std::string uri, std::string attribute, size_t cache_limit = 100000000) :

44 tiledb::Context ctx;
45 tiledb::ArraySchema schema(ctx, location);
46
47 if (!schema.has_attribute(attr)) {
48 throw std::runtime_error("no attribute '" + attr + "' is present in the TileDB array at '" + location + "'");
49 }
50
51 tiledb::Domain domain = schema.domain();
52 if (domain.ndim() != 2) {
53 throw std::runtime_error("TileDB array should have exactly two dimensions");
54 }
55
56 // We use 'int' for the domain, just in case the domain's absolute
57 // position exceeds Index_'s range, even if the actual range of the
58 // domain does not.
59 {
60 tiledb::Dimension dim = domain.dimension(0);
61 auto domain = dim.domain<int>();
62 first_offset = domain.first;
63 first_dim = domain.second - domain.first + 1;
64 first_tile = dim.tile_extent<int>();
65 }
66
67 {
68 tiledb::Dimension dim = domain.dimension(1);
69 auto domain = dim.domain<int>();
70 second_offset = domain.first;
71 second_dim = domain.second - domain.first + 1;
72 second_tile = dim.tile_extent<int>();
73 }

76 auto tiles_per_firstdim = static_cast<double>(second_dim) / second_tile;
77 auto tiles_per_seconddim = static_cast<double>(first_dim) / first_tile;
78 prefer_firstdim = tiles_per_firstdim <= tiles_per_seconddim;

191 template<bool accrow_, typename ExtractType_>
192 void extract_base(Index_ primary_start, Index_ primary_end, Value_* target, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
193 tiledb::Subarray subarray(work.ctx, work.array);
194
195 constexpr int dimdex = (accrow_ != transpose_);
196 auto primary_offset = (dimdex == 1 ? first_offset : second_offset);
197 subarray.add_range(1 - dimdex, primary_offset + primary_start, primary_offset + primary_end - 1);
198
199 // Adding ranges along the other dimension.
200 auto secondary_offset = (dimdex == 1 ? second_offset : first_offset);
201 constexpr bool indexed = std::is_same<ExtractType_, std::vector<Index_> >::value;
202 if constexpr(indexed) {
203 for (auto i : extract_value) {
204 auto adj = i + secondary_offset;
205 subarray.add_range(dimdex, adj, adj);
206 }
207 } else {
208 auto secondary_start = secondary_offset + extract_value;
209 subarray.add_range(dimdex, secondary_start, secondary_start + extract_length - 1);
210 }
211
212 tiledb::Query query(work.ctx, work.array);
213 query.set_subarray(subarray)
214 .set_layout(dimdex == 1 ? TILEDB_ROW_MAJOR : TILEDB_COL_MAJOR)
215 .set_data_buffer(attr, target, (primary_end - primary_start) * extract_length);
216
217 if (query.submit() != tiledb::Query::Status::COMPLETE) {
218 throw std::runtime_error("failed to read dense data from TileDB");
219 }
220 }

320 template<bool accrow_, typename ExtractType_>
321 const Value_* extract(Index_ i, Value_* buffer, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
322 // If there isn't any space for caching, we just extract directly.
323 if (work.num_chunks == 0) {
324 return extract_without_cache(i, buffer, extract_value, extract_length, work);
325 }
326
327 Index_ mydim = get_target_dim<accrow_>();
328 Index_ tile_mydim = get_target_chunk_dim<accrow_>();
329
330 const Value_* cache;
331 if (work.futurist) {
332 cache = extract_with_oracle(mydim, tile_mydim, extract_value, extract_length, work);
333 } else {
334 cache = extract_without_oracle(i, mydim, tile_mydim, extract_value, extract_length, work);
335 }
336
337 std::copy(cache, cache + extract_length, buffer);
338 return buffer;
339 }
